逻辑1 是否允许下单

if( 起点是郑州四环内 且 终点在郑州四环内 ){
  不允许 独享
  不允许 拼车
  不允许 代驾
}

if( (起点在郑州四环内 或 起点在郑州周边特殊价格区域内) 且 终点在登封市区){
  允许 独享 ，价格 看后文
  允许 拼车 ，价格 看后文
}

if(起点在登封市市区){
  允许 独享 ，价格 出租车计费标准
  允许 拼车 ，价格 35公里以内, 拼车价就是独享价/2，超过35公里？？，不允许拼车，只允许独享？？
  允许 代驾 ，代驾价格为 30 包10公里，之后每公里2块
  if( 终点在登封市区 ){
    允许 拼车 价格 35公里以内, 拼车价就是10元，
  }else{
    不允许 拼车
  }
}

if(起点在登封市乡镇){
  允许 独享 ，价格 出租车计费标准
  允许 拼车 ，价格 35公里以内, 拼车价就是独享价/2，超过35公里？？，不允许拼车，只允许独享？？
  允许 代驾 ，代驾价格为 30 包10公里，之后每公里2块
  if( 终点在登封市区 ){
    允许 拼车 价格 35公里以内, 拼车价就是10元，
  }else{
    不允许 拼车
  }
}




逻辑2 计算拼车价格和独享价格
伪代码：

if(终点在登封市区){
  if(起点 在 60元区域 ){
    拼车价格为60元 独享价格为180元
  }else if(起点 在 50元区域 ){
    拼车价格为50元 独享价格为160元
  }else if(起点 在 40元区域 ){
    拼车价格为40元 独享价格为120元
  }else if(起点 在 30元区域 ){
    拼车价格为30元 独享价格为100元
  }else if(起点 在 郑州四环内 且 起点 不在 郑州三环内 ){
    拼车价格为40元 独享价格为120元
  }else if(起点 在 郑州三环内 ){
    拼车价格为30元 独享价格为100元
  }
}
